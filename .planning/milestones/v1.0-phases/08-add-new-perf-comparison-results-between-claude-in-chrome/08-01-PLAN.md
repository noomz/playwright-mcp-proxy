Purpose: Demonstrate real-world performance trade-offs between three browser automation approaches — the proxy (metadata+diff), direct Playwright MCP (full payload), and claude-in-chrome (existing Chrome instance).     payload_bytes: int = 0
    estimated_tokens: int = 0

    name: str

    notes: list[str] = field(default_factory=list)

def _estimate_tokens(text: str) -> int:
    """Rough token estimate: ~4 chars per token for English text."""
